class DataSilos(Scene):
    def construct(self):
        # Create islands
        islands = VGroup(*[
            Circle(radius=1, fill_color=BLUE, fill_opacity=0.5)
            .add(Text(name).scale(0.5))
            for name in ["Confluence", "Jira", "Slack", "Drive", "GitHub"]
